65//模式( = CTRL1.PC )
66typedef enum {
67 GS_MODE_STAND_BY = 0,//(上电默认)低功耗待机,限制control3(CTRL3)寄存器控制位写入(包括自检启动[STC],温度测量启动[TCS],Force State下的手动触发[FRC]([FORCE]),详见手册),但不限制任何读取
68 GS_MODE_ACTIVE,//活跃,无限制读写,可以开始测量,可切换state选择测量方式(上电默认,需要外部命令来手动触发测量)
69 // GS_MODE_OFF,(掉电模式,此处不提供支持的模式)
70} GS_mode_t;
71
72//状态(在Active Mode下)( = CTRL1.FS )
73typedef enum {
74 GS_STATE_NORMAL = 0,//会自动根据配置触发测量
75 GS_STATE_FORCE,//(上电默认)需要设置control3(CTRL3)寄存器控制位STC来手动触发测量
76} GS_state_t;
77
78//数据量程( = CTRL4.RS )
79typedef enum {
80 GS_RANGE_OF_OUTPUT_14BIT = 0, //(上电默认) -8192LSB ~ +8191LSB ,1LSB = 0.15uT (量程: -8192*0.15uT ~ +8191*0.15uT)
81 GS_RANGE_OF_OUTPUT_15BIT, // -16384LSB ~ +16383LSB, 1LSB = 0.15uT (量程: -16384*0.15uT ~ +16383*0.15uT)
82} GS_range_of_output_t;

121}GS_angle_data_t;
