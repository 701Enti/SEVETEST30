64// 中断相关引脚的输出模式 SL2 SL1
65typedef enum
66{
67 BL5372_INTR_OUT_MODE0 = 0x00, // 闹钟A,闹钟B，周期性中断都从INTRA引脚输出
68 BL5372_INTR_OUT_MODE1 = 0x01, // INTRA引脚输出 闹钟A 周期性中断 /// INTRB引脚输出 闹钟B 32KHz时钟脉冲
69 BL5372_INTR_OUT_MODE2, // INTRA引脚输出 闹钟A,闹钟B /// INTRB引脚输出 周期性中断 32KHz时钟脉冲
70 BL5372_INTR_OUT_MODE3, // INTRA引脚输出 闹钟A /// INTRB引脚输出 闹钟A,闹钟B 周期性中断 32KHz时钟脉冲
71} BL5372_intr_out_mode_t;

73// 周期性中断(INT)模块的模式 CT2 CT1 CT0
74typedef enum
75{
76 BL5372_INT_MOD_MODE0 = 0x00, // 周期性中断接入的引脚为高电平(不输出)
77 BL5372_INT_MOD_MODE1 = 0x01, // 周期性中断接入的引脚为低电平
78 BL5372_INT_MOD_MODE2, // 周期性中断接入的引脚输出2Hz占空比50%的脉冲(脉冲模式)
79 BL5372_INT_MOD_MODE3, // 周期性中断接入的引脚输出1Hz占空比50%的脉冲(脉冲模式)
80 BL5372_INT_MOD_MODE4, // 当 秒 进位 周期性中断接入的引脚输出低电平
81 BL5372_INT_MOD_MODE5, // 当 分 进位 周期性中断接入的引脚输出低电平
82 BL5372_INT_MOD_MODE6, // 当 时 进位 周期性中断接入的引脚输出低电平
83 BL5372_INT_MOD_MODE7, // 当 月 进位 周期性中断接入的引脚输出低电平
84} BL5372_INT_module_mode_t;

86// 运行配置
87typedef struct BL5372_cfg_t
88{
89 //----------------控制寄存器1 BL5372_REG_CTRL_1
90 // D7-D0
91 bool alarm_A_en; // 设置为true将使能闹钟A AALE
92 bool alarm_B_en; // 设置为true将使能闹钟B BALE
93 BL5372_intr_out_mode_t intr_out_mode; // 中断输出模式,请查看BL5372_intr_out_mode_t SL2 SL1
94 bool test_en; // 设置为true进入测试模式 TEST
95 BL5372_INT_module_mode_t INT_module_mode; // 周期性中断模块的模式，请查看BL5372_INT_module_mode_t CT2 CT1 CT0
96
97 //----------------控制寄存器2 BL5372_REG_CTRL_2
98 // D5-D0
99 bool hour_24_clock_en; // 设置为true进行24小时制记时 12/24
100
101 // 写配置(只写ADJ)：设置为true启动正负30秒调整工作(秒复位00，如果原始秒处于30-59，分进一位),设置为flase表示希望进行正常工作
102 // 读配置(只读XSTP)：如果读出的值为true,RTC芯片处于停振检测标志，无法保证时间数据准确,但可能在正常走时
103 // 如果处于停振检测状态，XSL_ F6-F0 CT0-CT2 AALE BALE SL1-SL2 CLEN_ TEST 位复位为0 INTRA停止输出 INTRB输出32KHz脉冲
104 bool adj_en_or_xstp; // ADJ --- XSTP
105
106 bool out_32KHz_false; // 设置为true禁止32KHz脉冲输出 CLEN_
107
108 // 读配置(读写CTFG)：读出为true表示周期性中断输出进行中
109 // 写配置(读写CTFG)：设置为true允许保持现在的周期性中断输出(中断时默认为true,非中断设置为true无任何变化)
110 // 设置为false关闭现在的周期性中断输出(一次中断仅一次操作，不可逆)
111 bool INT_out_flag_or_out_keep; // CTFG
112
113 // 读配置(读写AAFG)：读出为true表示闹铃中断输出进行中
114 // 写配置(读写AAFG)：设置为true允许保持现在的闹铃中断输出(中断时默认为true,非中断为false设置为true无任何变化)
115 // 设置为false关闭现在的闹铃中断输出(一次中断仅一次操作，不可逆)
116 bool alarm_A_out_flag_or_out_keep; // AAFG
117
118 bool alarm_B_out_flag_or_out_keep; // BAFG 与alarm_A_out_flag_or_out_en相同但这是针对闹钟B的，读写BAFG
119
120} BL5372_cfg_t;
